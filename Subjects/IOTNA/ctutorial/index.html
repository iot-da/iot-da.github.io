void init_array(int array[], int size) { 
    int i;
    printf("Direccion de array: %p Sizeof array %lu \n", array, sizeof(array));
    for (i = 0; i < size; i++)
        array[i] = i;
    printf("Array initialized\n\n");
}

void init_array2(int array[N]) { 
    int i;
    printf("Direccion de array: %p Sizeof array %lu \n", array, sizeof(array));

        array[i] = i*2;
    printf("Array initialized\n\n");
}
